# Qiskit implementation snippet
def calculate_rate(B, P, h, sigma2, eps, l):
    sinr = P*abs(h)**2/sigma2
    shannon = log2(1+sinr)
    eta = 1 - (sigma2/(sigma2+P*abs(h)**2))**2
    zeta = log2(e)*qinv(eps)*sqrt(eta/l)
    return B*(shannon - zeta)

def utility(bitstring):

⚙️ FPGA kernel (HLS) – grover_iteration
typedef ap_fixed<16,8,AP_RND,AP_SAT> amp_t;
#define N_STATES 256
void grover_iteration(amp_t amps[N_STATES], bool good[N_STATES], amp_t out[N_STATES]) {
    #pragma HLS PIPELINE
    for(int i=0;i<N_STATES;i++) out[i] = good[i] ? -amps[i] : amps[i];
    amp_t sum=0; for(int i=0;i<N_STATES;i++) sum += out[i];
    amp_t mean = sum / N_STATES;
    for(int i=0;i<N_STATES;i++) out[i] = (2*mean) - out[i]; // diffusion
}

Parallel factor 16 → ~192 cycles @200MHz → <10 µs per full Grover
